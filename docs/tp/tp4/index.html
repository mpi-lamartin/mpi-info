let g0 = { 
    n_variables = 5;
    regles1 = [(0, 'b'); (1, 'a'); (2, 'b'); (4, 'a')];
    regles2 = [(0, 1, 2); (0, 2, 1); (0, 3, 1); (1, 1, 4); (3, 1, 2)];
    epsilon = false
}

let init g u =
    let n = String.length u in
    let k = g.n_variables in
    let t = Array.make (n+1) [||] in
    for i = 0 to n do
        t.(i) <- Array.make_matrix n k false
    done;
    if g.epsilon then t.(0).(0).(0) <- true;

    let rec aux = function
        | [] -> ()
        | (i, a)::q ->
            for d = 0 to n - 1 do
                if u.[d] = a then t.(1).(d).(i) <- true
            done;
            aux q
    in
    aux g.regles1;
    t

(* 7 *)
let cyk g u =
    let t = init g u in
    let n = String.length u in

    for l = 2 to n do
        for d = 0 to n - l do
            let rec aux = function
                | [] -> ()
                | (i, j, k)::q ->
                    for p = 1 to l - 1 do
                        if d + p < n && t.(p).(d).(j) && t.(l-p).(d+p).(k) 
                        then t.(l).(d).(i) <- true
                    done;
                    aux q
            in
            aux g.regles2
        done
    done;
    t.(n).(0).(0);;

type arbre = Vide | Un of int * char | Deux of int * arbre * arbre

let arbre_derivation g u : arbre option =
    let n = String.length u in
    let k = g.n_variables in
    let t = Array.make (n+1) [||] in
    for i = 0 to n do
        t.(i) <- Array.make_matrix n k None
    done;
    if g.epsilon then t.(0).(0).(0) <- Some Vide;
    let rec aux = function
        | [] -> ()
        | (i, a)::q ->
            for d = 0 to n - 1 do
                if u.[d] = a then t.(1).(d).(i) <- Some (Un (i, a))
            done;
            aux q
    in
    aux g.regles1;

    for l = 2 to n do
        for d = 0 to n - l do
            let rec aux = function
                | [] -> ()
                | (i, j, k)::q ->
                    for p = 1 to l - 1 do
                        if d + p < n
                        then match t.(p).(d).(j), t.(l-p).(d+p).(k) with
                        | Some a1, Some a2 -> t.(l).(d).(i) <- Some (Deux (i, a1, a2))
                        | _ -> ()
                    done;
                    aux q
            in
            aux g.regles2
        done
    done;
    t.(n).(0).(0);;

(* 10 *)

arbre_derivation g0 "aab";;

L'algorithme CYK permet de déterminer, par programmation dynamique, si un mot est généré par une grammaire non contextuelle.
Il permet aussi de trouver un arbre de dérivation pour ce mot, par analyse syntaxique ascendante (bottom-up : des feuilles à la racine).

L'algorithme prend en entrée une grammaire G=(Σ,V,R,S)G = (\Sigma, V, R, S)G=(Σ,V,R,S) en forme normale de Chomsky, c'est-à-dire que toutes les règles sont de la forme X→YZX \to YZX→YZ ou X→aX \to aX→a ou S→ϵS \to \epsilonS→ϵ, où X,Y,Z∈VX, Y, Z \in VX,Y,Z∈V, Y≠S,Z≠SY \neq S, Z\neq SY=S,Z=S et a∈Σa \in \Sigmaa∈Σ.
On note V={X0,X1,…,Xk−1}V =\{X_0, X_1, \ldots, X_{k-1}\}V={X0​,X1​,…,Xk−1​} les variables, où S=X0S = X_0S=X0​.

Arbre de dérivation​

On définit le type suivant pour modéliser un arbre de dérivation :

type arbre = Vide | Un of int * char | Deux of int * arbre * arbre

où :

  • Vide correspond à une dérivation S→ϵS \to \epsilonS→ϵ,
  • Un (i, a) correspond à une dérivation Xi→aX_i \to aXi​→a,
  • Deux (i, j, k) correspond à une dérivation Xi→XjXkX_i \to X_j X_kXi​→Xj​Xk​.

On rappelle que le type option est défini nativement en OCaml par :

type 'a option = None | Some of 'a

On adapte alors l'algorithme CYK avec t : arbre option array array array où t[l][d][i] contient un arbre de dérivation pour Xi⇒∗ud…ud+l−1X_i \Rightarrow^* u_{d} \ldots u_{d + l - 1}Xi​⇒∗ud​…ud+l−1​ s'il en existe un, None sinon.
